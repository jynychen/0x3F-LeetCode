                g[x].append((y, w))
                g[y].append((x, w))

            depth = [0] * n
            pa = [[-1] * n for _ in range(m)]

                pa[0][x] = fa
                for y, w in g[x]:

                        depth[y] = depth[x] + 1
                        dis[y] = dis[x] + w
                        dfs(y, x)
            dfs(0, -1)

            for i in range(m - 1):
