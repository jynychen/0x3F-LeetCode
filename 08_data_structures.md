    class LazyHeap:
        def __init__(self):

        # delete
        def remove(self, x: Any) -> None:
            self.remove_cnt[x] += 1  # lazy deletion
            self.size -= 1

        def _apply_remove(self) -> None:
            while self.heap and self.remove_cnt[self.heap[0]] > 0:
                self.remove_cnt[self.heap[0]] -= 1
                heappop(self.heap)

        def top(self) -> Any:
            self._apply_remove()

        # pop from heap
        def pop(self) -> Any:
            self._apply_remove()
            self.size -= 1
            return heappop(self.heap)

        # push to heap
        def push(self, x: Any) -> None:
            if self.remove_cnt[x] > 0:
            else:
                heappush(self.heap, x)
            self.size += 1

    class UnionFind:
        def __init__(self, n: int):
            self.fa = list(range(n))  # representative

        def find(self, x: int) -> int:
            fa = self.fa
            if fa[x] != x:
                root = self.find(fa[x])
                fa[x] = root
            return fa[x]

        def same(self, x: int, y: int) -> bool:
            return self.find(x) == self.find(y)

        def get_relative_distance(self, from_: int, to: int) -> int:
            self.find(from_)
            self.find(to)
            # to-from = (x-from) - (x-to) = dis[from] - dis[to]
            return self.dis[from_] - self.dis[to]

        def merge(self, from_: int, to: int, value: int) -> bool:
            x, y = self.find(from_), self.find(to)
            dis = self.dis
                # to-from = (x-from) - (x-to) = dis[from] - dis[to] = value
                return dis[from_] - dis[to] == value
            #    x --------- y
            #   /           /
            # from ------- to
            # since y-from = (y-x) + (x-from) = (y-to) + (to-from)
            # so y-x = (to-from) + (y-to) - (x-from) = value + dis[to] - dis[from]
            self.fa[x] = y
            return True

    class SparseTable:
        def __init__(self, nums: List[int], op: Callable[[int, int], int]):
            n = len(nums)
            w = n.bit_length()
            st = [[0] * n for _ in range(w)]
            st[0] = nums[:]
            for i in range(1, w):
                for j in range(n - (1 << i) + 1):
                    st[i][j] = op(st[i - 1][j], st[i - 1][j + (1 << (i - 1))])
            self.st = st
            self.op = op

        # return op(nums[l:r])
        def query(self, l: int, r: int) -> int:
            k = (r - l).bit_length() - 1
            return self.op(self.st[k][l], self.st[k][r - (1 << k)])
